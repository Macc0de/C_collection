// Подсчет общего количества гласных и согласных в строке
// (1)
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
#include <ctype.h>

void func(char* msg)
{
	int vowels = 0;
	int consonants = 0;
	int i = 0;
	for (i = 0; msg[i] != '\0'; i++) {
		if ((tolower(msg[i]) == 'b') || (tolower(msg[i]) == 'c') || (tolower(msg[i]) == 'd') || (tolower(msg[i]) == 'f') || (tolower(msg[i]) == 'g') || (tolower(msg[i]) == 'h') || (tolower(msg[i]) == 'j') || (tolower(msg[i]) == 'k') || (tolower(msg[i]) == 'l') || (tolower(msg[i]) == 'm') || (tolower(msg[i]) == 'n') || (tolower(msg[i]) == 'p') || (tolower(msg[i]) == 'q') || (tolower(msg[i]) == 'r') || (tolower(msg[i]) == 's') || (tolower(msg[i]) == 't') || (tolower(msg[i]) == 'v') || (tolower(msg[i]) == 'w') || (tolower(msg[i]) == 'x') || (tolower(msg[i]) == 'y') || (tolower(msg[i]) == 'z')) {
			consonants++;
		}
		else if ((tolower(msg[i]) == 'a') || (tolower(msg[i]) == 'e') || (tolower(msg[i]) == 'o') || (tolower(msg[i]) == 'u') || (tolower(msg[i]) == 'y') || (tolower(msg[i]) == 'i')) {
			vowels++;
		}
	}
	printf("Total score: %d %d", consonants, vowels);
}

int main()
{
	char msg[30];
	fgets(msg, 30, stdin);
	func(msg);
	return 0;
}
